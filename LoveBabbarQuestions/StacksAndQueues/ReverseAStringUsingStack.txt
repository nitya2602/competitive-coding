char* reverse(char *S, int len)
{
    //code here
    stack<int> stack;
    for(int i = 0; i < len; i++)
    {
        stack.push(S[i]);
    }
    char* ans = (char*)malloc(sizeof(char)*(len+1));
    int i = 0;
    while(!stack.empty())
    {
        ans[i++] = stack.top();
        stack.pop();
    }
    ans[i] = '\0';
    return ans;
}
