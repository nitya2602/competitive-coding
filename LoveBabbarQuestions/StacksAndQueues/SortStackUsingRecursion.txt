void SortedStack :: sortedInsert(int x)
{
    if(s.empty() || x > s.top())
    {
        s.push(x);
    }
    else
    {
        int temp = s.top();
        sortedInsert(x);
        s.push(temp);
    }
}
void SortedStack :: sort()
{
   //Your code here
   if(!s.empty())
   {
       int x = s.top();
       s.pop();
       sort();
       sortedInsert(x);
   }
}
