#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    std::cin>>t;
    while(t--)
    {
        int w, h, n;
        std::cin>>w>>h>>n;
        vector<int> a, b;
        a.push_back(0);
        b.push_back(0);
        for(int i = 0; i < n; i++)
        {
            int x, y;
            std::cin>>x>>y;
            a.push_back(x);
            b.push_back(y);
        }
        a.push_back(w+1);
        b.push_back(h+1);
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());
        int mh = 0, mw = 0;
        for(int i = 0; i < a.size()-1; i++)
        {
            mw = max(mw, a[i+1]-a[i]-1);
            mh = max(mh, b[i+1]-b[i]-1);
        }
        std::cout<<mw*mh<<" ";
    }
    return 0;
}
