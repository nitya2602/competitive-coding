//O(NlogN) time complexity and O(1) space

class Solution{
    public:
    //Function to find the minimum number of platforms required at the
    //railway station such that no train waits.
    int findPlatform(int arr[], int dep[], int n)
    {
    	// Your code here
    	sort(arr, arr+n);
    	sort(dep, dep+n);
    	int i = 0, j = 0;
    	int plat = 0, result = 0;
    	while(i < n && j < n)
    	{
    	    if(arr[i] <= dep[j])
    	    {
    	        i++;
    	        plat++;
    	    }
    	    else if(arr[i] > dep[j])
    	    {
    	        j++;
    	        plat--;
    	    }
    	    if(plat > result)
    	    {
    	        result = plat;
    	    }
    	}
    	return result;
    }
};
