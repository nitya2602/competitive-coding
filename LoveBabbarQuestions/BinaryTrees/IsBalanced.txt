int postorder(Node* root, bool &flag)
{
    if(root == NULL)
    {
        return 0;
    }
    int left = postorder(root -> left, flag);
    int right = postorder(root -> right, flag);
    if(abs(left - right) > 1)
    {
        flag = false;
    }
    return max(left, right) + 1;
}
