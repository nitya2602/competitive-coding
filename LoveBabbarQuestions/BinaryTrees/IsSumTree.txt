class Solution
{
    public:
    int postorder(Node* root, bool &flag)
    {
        if(root == NULL)
        {
            return 0;
        }
        int left = postorder(root -> left, flag);
        int right = postorder(root -> right, flag);
        if(!(root->left==NULL && root->right==NULL) && root -> data != left + right)
        {
            flag = false;
        }
        return left + right + root -> data;
    }
    bool isSumTree(Node* root)
    {
         // Your code here
         bool flag = true;
         postorder(root, flag);
         return flag;
    }
};
