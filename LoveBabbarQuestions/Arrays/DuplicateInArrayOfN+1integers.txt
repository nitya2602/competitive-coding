class Solution {
public:
    int findDuplicate(vector<int>& nums) {
        int tortoise = nums[0];
        int hare = nums[0];
        do{
            tortoise = nums[tortoise];
            hare = nums[nums[hare]];
        }while(hare != tortoise);
        
        tortoise = nums[0];
        while(tortoise != hare)
        {
            hare = nums[hare];
            tortoise = nums[tortoise];
        }
        return hare;
    }
};
