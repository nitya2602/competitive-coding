class Solution{
    public:
    #define MOD 1000000007
    long long countWays(int n, int k){
        // code here
        if(n == 0)
            return 0;
        if(n == 1)
            return k;
        long long same = k;
        long long diff = k*(k-1);
        for(int i = 3; i <= n; i++)
        {
            int prevDiff = diff;
            diff = (diff + same)*(k-1)%MOD;
            same = prevDiff%MOD;
        }
        return (same + diff)%MOD;
    }
};
