//only 2 possibilities - friend remains single, friend pairs up

class Solution
{
public:
    #define MOD 1000000007
    int countFriendsPairings(int n) 
    { 
        // code here
        long long dp[n+1];
        for(int i = 0; i <= n; i++)
        {
            if(i <= 2)
                dp[i] = i;
            else
            {
                dp[i] = (dp[i-1]%MOD + (i - 1)*dp[i - 2]%MOD)%MOD;
            }
        }
        return dp[n];
    }
};
